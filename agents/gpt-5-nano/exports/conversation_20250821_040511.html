python import pygame import sys import random # Initialize Pygame pygame.init() # Screen settings CELL_SIZE = 20 GRID_WIDTH = 31 # number of cells horizontally GRID_HEIGHT = 23 # number of cells vertically WIDTH = GRID_WIDTH * CELL_SIZE HEIGHT = GRID_HEIGHT * CELL_SIZE # Colors BG_COLOR = (40, 40, 40) SNAKE_COLOR = (0, 255, 0) SNAKE_HEAD_COLOR = (0, 200, 0) FOOD_COLOR = (255, 0, 0) GRID_COLOR = (50, 50, 50) TEXT_COLOR = (255, 255, 255) # Directions UP = (0, -1) DOWN = (0, 1) LEFT = (-1, 0) RIGHT = (1, 0) # Initialize screen screen = pygame.display.set_mode((WIDTH, HEIGHT)) pygame.display.set_caption("Snake Game") # Clock for controlling speed clock = pygame.time.Clock() FONT = pygame.font.SysFont(None, 36) def random_food_position(snake): positions = set(snake) while True: x = random.randint(0, GRID_WIDTH - 1) y = random.randint(0, GRID_HEIGHT - 1) if (x, y) not in positions: return (x, y) def draw_grid(): for x in range(0, WIDTH, CELL_SIZE): pygame.draw.line(screen, GRID_COLOR, (x, 0), (x, HEIGHT)) for y in range(0, HEIGHT, CELL_SIZE): pygame.draw.line(screen, GRID_COLOR, (0, y), (WIDTH, y)) def draw_snake(snake): for i, (x, y) in enumerate(snake): rect = pygame.Rect(x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE, CELL_SIZE) color = SNAKE_HEAD_COLOR if i == 0 else SNAKE_COLOR pygame.draw.rect(screen, color, rect) def draw_food(pos): x, y = pos rect = pygame.Rect(x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE, CELL_SIZE) pygame.draw.rect(screen, FOOD_COLOR, rect) def show_text(text, pos): img = FONT.render(text, True, TEXT_COLOR) screen.blit(img, pos) def main(): # Initial snake in the middle start_x = GRID_WIDTH // 2 start_y = GRID_HEIGHT // 2 snake = [(start_x, start_y), (start_x - 1, start_y), (start_x - 2, start_y)] direction = RIGHT next_direction = RIGHT food = random_food_position(snake) score = 0 speed = 10 # initial frames per second game_over = False while True: for event in pygame.event.get(): if event.type == pygame.QUIT: pygame.quit() sys.exit() elif event.type == pygame.KEYDOWN: if event.key in (pygame.K_w, pygame.K_UP): if direction != DOWN: next_direction = UP elif event.key in (pygame.K_s, pygame.K_DOWN): if direction != UP: next_direction = DOWN elif event.key in (pygame.K_a, pygame.K_LEFT): if direction != RIGHT: next_direction = LEFT elif event.key in (pygame.K_d, pygame.K_RIGHT): if direction != LEFT: next_direction = RIGHT elif event.key == pygame.K_ESCAPE: pygame.quit() sys.exit() elif event.key == pygame.K_r and game_over: return main() # restart if not game_over: direction = next_direction # Compute new head head_x, head_y = snake[0] dx, dy = direction new_head = (head_x + dx, head_y + dy) # Check wall collision if (new_head[0] < 0 or new_head[0] >= GRID_WIDTH or new_head[1] < 0 or new_head[1] >= GRID_HEIGHT): game_over = True else: # Check self-collision if new_head in snake: game_over = True else: # Move snake snake.insert(0, new_head) # Check food if new_head == food: score += 1 speed = 10 + score # increase speed with score food = random_food_position(snake) else: snake.pop() # remove tail # Rendering screen.fill(BG_COLOR) draw_grid() draw_food(food) draw_snake(snake) # HUD show_text(f"Score: {score}", (10, 10)) if game_over: show_text("Game Over! Press R to restart.", (WIDTH // 2 - 170, HEIGHT // 2 - 10)) pygame.display.flip() clock.tick(speed) if __name__ == "__main__": main()
