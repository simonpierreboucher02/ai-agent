```python
import pygame
import sys
import random

# Initialize Pygame
pygame.init()

# Screen settings
CELL_SIZE = 20
GRID_WIDTH = 31  # number of cells horizontally
GRID_HEIGHT = 23  # number of cells vertically
WIDTH = GRID_WIDTH * CELL_SIZE
HEIGHT = GRID_HEIGHT * CELL_SIZE

# Colors
BG_COLOR = (40, 40, 40)
SNAKE_COLOR = (0, 255, 0)
SNAKE_HEAD_COLOR = (0, 200, 0)
FOOD_COLOR = (255, 0, 0)
GRID_COLOR = (50, 50, 50)
TEXT_COLOR = (255, 255, 255)

# Directions
UP = (0, -1)
DOWN = (0, 1)
LEFT = (-1, 0)
RIGHT = (1, 0)

# Initialize screen
screen = pygame.display.set_mode((WIDTH, HEIGHT))
pygame.display.set_caption("Snake Game")

# Clock for controlling speed
clock = pygame.time.Clock()
FONT = pygame.font.SysFont(None, 36)

def random_food_position(snake):
    positions = set(snake)
    while True:
        x = random.randint(0, GRID_WIDTH - 1)
        y = random.randint(0, GRID_HEIGHT - 1)
        if (x, y) not in positions:
            return (x, y)

def draw_grid():
    for x in range(0, WIDTH, CELL_SIZE):
        pygame.draw.line(screen, GRID_COLOR, (x, 0), (x, HEIGHT))
    for y in range(0, HEIGHT, CELL_SIZE):
        pygame.draw.line(screen, GRID_COLOR, (0, y), (WIDTH, y))

def draw_snake(snake):
    for i, (x, y) in enumerate(snake):
        rect = pygame.Rect(x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE, CELL_SIZE)
        color = SNAKE_HEAD_COLOR if i == 0 else SNAKE_COLOR
        pygame.draw.rect(screen, color, rect)

def draw_food(pos):
    x, y = pos
    rect = pygame.Rect(x * CELL_SIZE, y * CELL_SIZE, CELL_SIZE, CELL_SIZE)
    pygame.draw.rect(screen, FOOD_COLOR, rect)

def show_text(text, pos):
    img = FONT.render(text, True, TEXT_COLOR)
    screen.blit(img, pos)

def main():
    # Initial snake in the middle
    start_x = GRID_WIDTH // 2
    start_y = GRID_HEIGHT // 2
    snake = [(start_x, start_y),
             (start_x - 1, start_y),
             (start_x - 2, start_y)]
    direction = RIGHT
    next_direction = RIGHT
    food = random_food_position(snake)
    score = 0
    speed = 10  # initial frames per second
    game_over = False

    while True:
        for event in pygame.event.get():
            if event.type == pygame.QUIT:
                pygame.quit()
                sys.exit()
            elif event.type == pygame.KEYDOWN:
                if event.key in (pygame.K_w, pygame.K_UP):
                    if direction != DOWN:
                        next_direction = UP
                elif event.key in (pygame.K_s, pygame.K_DOWN):
                    if direction != UP:
                        next_direction = DOWN
                elif event.key in (pygame.K_a, pygame.K_LEFT):
                    if direction != RIGHT:
                        next_direction = LEFT
                elif event.key in (pygame.K_d, pygame.K_RIGHT):
                    if direction != LEFT:
                        next_direction = RIGHT
                elif event.key == pygame.K_ESCAPE:
                    pygame.quit()
                    sys.exit()
                elif event.key == pygame.K_r and game_over:
                    return main()  # restart

        if not game_over:
            direction = next_direction

            # Compute new head
            head_x, head_y = snake[0]
            dx, dy = direction
            new_head = (head_x + dx, head_y + dy)

            # Check wall collision
            if (new_head[0] < 0 or new_head[0] >= GRID_WIDTH or
                new_head[1] < 0 or new_head[1] >= GRID_HEIGHT):
                game_over = True
            else:
                # Check self-collision
                if new_head in snake:
                    game_over = True
                else:
                    # Move snake
                    snake.insert(0, new_head)

                    # Check food
                    if new_head == food:
                        score += 1
                        speed = 10 + score  # increase speed with score
                        food = random_food_position(snake)
                    else:
                        snake.pop()  # remove tail

        # Rendering
        screen.fill(BG_COLOR)
        draw_grid()
        draw_food(food)
        draw_snake(snake)

        # HUD
        show_text(f"Score: {score}", (10, 10))
        if game_over:
            show_text("Game Over! Press R to restart.", (WIDTH // 2 - 170, HEIGHT // 2 - 10))

        pygame.display.flip()
        clock.tick(speed)

if __name__ == "__main__":
    main()
```

How to play:
- Arrow keys or WASD to move
- Avoid hitting walls or yourself
- Eat red blocks (food) to grow and increase score
- Press R to restart after game over
- Press Esc to quit

```r
# snake_game.R
library(shiny)

# Game constants
GRID_WIDTH <- 31
GRID_HEIGHT <- 23
CELL_SIZE <- 20  # used for drawing scale in plotting

# UI
ui <- fluidPage(
  tags$head(
    # JavaScript to capture keyboard input and send to Shiny
    tags$script(HTML("
      // Map keys to directions
      const keyMap = {
        'ArrowUp': 'UP',
        'ArrowDown': 'DOWN',
        'ArrowLeft': 'LEFT',
        'ArrowRight': 'RIGHT',
        'w': 'UP',
        'W': 'UP',
        's': 'DOWN',
        'S': 'DOWN',
        'a': 'LEFT',
        'A': 'LEFT',
        'd': 'RIGHT',
        'D': 'RIGHT'
      };
      $(document).keydown(function(e) {
        const key = e.key;
        if (keyMap[key] !== undefined) {
          Shiny.setInputValue('dir', keyMap[key], {priority: 'event'});
        }
        if (key === 'Escape') {
          // optional: stop the app or perform another action
        }
      });
    "))
  ),
  titlePanel("Snake Game (R + Shiny)"),
  fluidRow(
    column(width = 9,
           plotOutput("gamePlot", height = GRID_HEIGHT * 20, width = GRID_WIDTH * 20)
    ),
    column(width = 3,
           wellPanel(
             h4("Controls"),
             p("Use Arrow keys or WASD to move."),
             br(),
             p("Score is shown on the canvas."),
             br(),
             p("Game over: press R to restart.")
           ),
           br(),
           verbatimTextOutput("hud")
    )
  ),
  # Hidden text to capture restart key (R)
  tags$script(HTML("
    // Allow pressing R to restart when game over
    $(document).keydown(function(e){
      if (e.key === 'r' || e.key === 'R') {
        Shiny.setInputValue('restart', '1', {priority: 'event'});
      }
    });
  "))
)

# Server
server <- function(input, output, session) {

  # Game state stored as reactive values
  state <- reactiveValues(
    snake = NULL,       # list of (x, y) pairs, head first
    dir = NULL,         # current direction
    nextDir = NULL,     # next direction (to apply after input)
    food = NULL,          # position (x, y)
    score = 0,
    speed = 100,          # ms per move
    gameOver = FALSE
  )

  # Initialize game
  init_game <- function() {
    midX <- GRID_WIDTH %/% 2
    midY <- GRID_HEIGHT %/% 2
    state$snake <- list(c(midX, midY),
                        c(midX - 1, midY),
                        c(midX - 2, midY))
    state$dir <- c(1, 0)       # RIGHT
    state$nextDir <- c(1, 0)
    state$food <- random_food(state$snake)
    state$score <- 0
    state$speed <- 100
    state$gameOver <- FALSE
  }

  # Random food not on the snake
  random_food <- function(snake) {
    snake_set <- do.call(rbind, snake)
    repeat {
      x <- sample(0:(GRID_WIDTH-1), 1)
      y <- sample(0:(GRID_HEIGHT-1), 1)
      if (!any(snake_set[,1] == x & snake_set[,2] == y)) {
        return(c(x, y))
      }
    }
  }

  # Initialize on startup
  observe({
    init_game()
  })

  # Handle direction input from JS
  observeEvent(input$dir, {
    if (is.null(input$dir)) return()
    # convert to vector
    d <- switch(input$dir,
                "UP" = c(0, -1),
                "DOWN" = c(0, 1),
                "LEFT" = c(-1, 0),
                "RIGHT" = c(1, 0),
                NULL)
    if (is.null(d)) return()
    # prevent reversing into self
    cur <- state$dir
    if (!is.null(cur) && (d[1] == -cur[1] && d[2] == -cur[2])) {
      return()
    }
    state$nextDir <- d
  })

  # Restart on 'R' press
  observeEvent(input$restart, {
    init_game()
  })

  # Main game loop
  timer <- reactiveTimer(state$speed)

  observe({
    timer()
    if (state$gameOver) return()
    # Apply direction
    state$dir <- state$nextDir

    # Compute new head
    head <- state$snake[[1]]
    new_head <- c(head[1] + state$dir[1], head[2] + state$dir[2])

    # Wall collision
    if (new_head[1] < 0 || new_head[1] >= GRID_WIDTH ||
        new_head[2] < 0 || new_head[2] >= GRID_HEIGHT) {
      state$gameOver <- TRUE
      return()
    }

    # Self-collision
    body <- do.call(rbind, state$snake)
    if (any(apply(body, 1, function(p) all(p == new_head)))) {
      state$gameOver <- TRUE
      return()
    }

    # Move snake
    state$snake <- c(list(new_head), state$snake)

    # Food
    if (all(new_head == state$food)) {
      state$score <- state$score + 1
      state$speed <- max(40, state$speed - 3)  # speed up a bit
      state$food <- random_food(state$snake)
    } else {
      state$snake <- state$snake[-length(state$snake)]
    }
  })

  # Rendering
  output$gamePlot <- renderPlot({
    # Plot empty grid
    plot(NA, xlim = c(0, GRID_WIDTH), ylim = c(0, GRID_HEIGHT),
         xaxt = 'n', yaxt = 'n', xlab = '', ylab = '',
         type = 'n', bty = 'n', asp = 1)

    # Grid lines
    abline(h = seq(0, GRID_HEIGHT, by = 1), col = "grey90", lwd = 1)
    abline(v = seq(0, GRID_WIDTH, by = 1), col = "grey90", lwd = 1)

    # Draw food
    f <- state$food
    if (!is.null(f)) {
      rect(f[1], f[2], f[1] + 1, f[2] + 1, col = "red", border = NA)
    }

    # Draw snake
    if (!is.null(state$snake)) {
      for (i in seq_along(state$snake)) {
        s <- state$snake[[i]]
        col <- if (i == 1) "darkgreen" else "green"
        rect(s[1], s[2], s[1] + 1, s[2] + 1, col = col, border = NA)
      }
    }

    # Title overlay-like HUD
    title <- paste("Score:", state$score, sep = " ")
    mtext(title, side = 3, line = -1, adj = 0.0, cex = 1.0)

    # If game over, show message
    if (state$gameOver) {
      mtext("Game Over! Press R to restart.", side = 3, line = -2, adj = 0.5, cex = 1.2, font = 2)
    }
  }, res = 96)

  # HUD text
  output$hud <- renderText({
    paste0("Score: ", state$score, ifelse(state$gameOver, "  (Game Over)", ""))
  })

}

# Run the app
shinyApp(ui = ui, server = server)
```
